    ```python
    def to_line(self) -> str:
        parts = []
        if self.is_important:
            parts.append("[IMPORTANT]")
        if self.is_popular:
            parts.append("[POPULAR]")
        parts.append(f"{self.author}: {self.content}")
        if self.reaction_count >= 5:
            parts.append(f"[{self.reaction_count} reactions]")
        if self.embeds_text:
            parts.append("(" + "; ".join(self.embeds_text) + ")")
        return " ".join(parts)
    ```

    ```python
    def classify_attachment(attachment: discord.Attachment) -> str:
        ct = attachment.content_type or ""
        if ct.startswith("image/"):
            return "image"
        elif ct.startswith("video/"):
            return "video"
        elif ct.startswith("audio/"):
            return "audio"
        return "file"
    ```

    Add `extract_embed_text()` function:

    ```python
    def extract_embed_text(embed: discord.Embed) -> str | None:
        parts = []
        if embed.title:
            parts.append(embed.title)
        if embed.description:
            desc = embed.description[:200] if len(embed.description) > 200 else embed.description
            parts.append(desc)
        return " - ".join(parts) if parts else None
    ```
